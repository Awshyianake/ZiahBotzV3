```bash

pkg install nodejs
pkg install libwebp
pkg install ffmpeg
pkg install imagemagick
pkg install yarn
pkg install git
git clone https://github.com/Ronzz-Ofc/ZiahBotzV3
termux-setup-storage
cd /sdcard
cd ZiahBotzV3
yarn install
npm start
```
